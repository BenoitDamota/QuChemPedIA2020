  3 /**
  4  * Initialize the drawer to draw to canvas
  5  */
  6 function drawCanvas() {
  7   var array_canvas = document.querySelectorAll("canvas");
  8 
  9   array_canvas.forEach(function (element) {
 10     let smileDrawer = new SmilesDrawer.Drawer({
 11       width: 150,
 12       height: 150,
 13     });
 14 
 15     // Smile value is display in hidden div : '"smi_value"_smile' and as html :"smi_value"
 16     var input = String($(element).attr("id").replace("_canvas", "")) + "_smile";
 17     var smile = document.getElementById(input);
 18 
 19     SmilesDrawer.parse(
 20       $(smile).html(),
 21       function (tree) {
 22         smileDrawer.draw(tree, $(element).attr("id"), "light", false);
 23       },
 24       function (err) {
 25         //console.log(err); // Display error if needed
 26       }
 27     );
 28   });
 29 }
