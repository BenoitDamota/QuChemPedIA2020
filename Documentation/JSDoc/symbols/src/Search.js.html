  5  * To display a loader during the initiation of datatable

  8 function loading(isTrue) {

 10     var div_flex = document.createElement("div");

 12     $(div_flex).attr("id", "loading");
 13     $(div_flex).css("margin-top", "30px");
 14 
 15     var spinner = document.createElement("div");
 16     $(spinner).attr("class", "spinner-border");
 17     $(spinner).attr("role", "status");
 18     $(spinner).attr("style", "width: 3rem; height: 3rem;");
 19 
 20     var span_spinner = document.createElement("span");
 21     $(span_spinner).attr("class", "sr-only");
 22     $(span_spinner).html("Loading...");
 23 
 24     $(spinner).append($(span_spinner));
 25     $(div_flex).append($(spinner));
 26 
 27     $("#result_number").append($(div_flex));
 28   } else {
 29     var div_flex = document.getElementById("loading");
 30     $(div_flex).remove();
 31   }

 35 /** 
 36  * Function to prevent click, when user want to highlight
 37  */ 
 38 function getSelected() {
 39   if (window.getSelection) {
 40     return window.getSelection().toString();
 41   } else if (document.getSelection) {
 42     return document.getSelection().toString();
 43   } else {
 44     var selection = document.selection && document.selection.createRange();
 45     if (selection.text) {
 46       return selection.text.toString();
 47     }
 48     return "";
 49   }
 50   return "";
 51 }
 52 /**
 53  * To add a href click to /details/id
 54  */ 
 55 function addClick() {
 56   var all_li = document.querySelectorAll("#display_result > li");
 57   for (let i = 0; i < all_li.length; ++i) {
 58     all_li[i].style.cursor = "pointer";
 59     all_li[i].addEventListener("click", function () {
 60       var selection = getSelected().toString();
 61       //When user highlight, if nothing highlight -> go to detail page
 62       if (selection === "") {
 63         location.href = "/QuChemPedIA/html/page_detail.html?id=" + all_li[i].id;
 64       }
 65     });
 66   }
 67 }

 83 /**
 84  *  Search function call after each click on submit button <br/>
 85  *  First we check if it's the first request make from index.html <br/>
 86  *  YES -> Creation of navbar, and each div use to display result,          
 87  *  pagination                                         <br/>
 88  *  NO -> We clear the area to display the new request <br/>                                                      
 89  *  @param {int} page_number - page number to display                                    
 90  *  @param {int} entrie_page - entries per page                                           
 91  *                                                                         
 92  */
 93 function search(page_number, entrie_page) {
 94   if ($("#query").val() != "") {
 95     // First submit, we create navbar
 96     if ($("#navbar_top").length === 0) {
 97       //init value for entrie_page -> default 25 entries per page or other value if we dont search by URL
 98       let url = new URL(document.location.href);
 99       let entrie = url.searchParams.get("showresult");
100       if (entrie === null) {
101         entrie_page = 25;
102       } else entrie_page = entrie;
103 
104       var navbar = document.createElement("nav");
105       $(navbar).attr("id", "navbar_top");
106       $(navbar).addClass("navbar navbar-expand-lg navbar-light bg-light");
107 
108       var logo = document.createElement("a");
109       $(logo).attr("id", "home_button");
110       $(logo).attr("href", "index.html");
111       $(logo).html(
112         'QuChemPedIA'
113       );
114       $(logo).addClass("navbar-brand primary");
115       $(navbar).append(logo);
116 
117       var navbarSupportedContent = document.createElement("div");
118       $(navbarSupportedContent).addClass("collapse navbar-collapse");
119       $(navbar).append(navbarSupportedContent);
120 
121       $(document.body).append(navbar);
122 
123       var search_bar = document.getElementById("search_form");
124       var search_bar_prime = search_bar.cloneNode(true);
125       $(search_bar_prime).removeClass("w-100 justofy-content-center");
126       $(navbar).append($(search_bar_prime));
127 
128       var button_submit = document.getElementById("submit_search");
129       $(button_submit)
130         .removeClass(
131           "mt-1 mb-1 mt-lg-0 btn btn-info default-primary-color col-12 col-lg-2"
132         )
133         .addClass("btn btn-primary mt-lg-0  my-2 my-sm-0");
134 
135       var div_result_number = document.createElement("div");
136       $(div_result_number).attr("id", "result_number");
137       $(div_result_number).addClass("");
138       $(div_result_number).css("display", "flex");
139       $(div_result_number).css("justify-content", "center");
140       $(div_result_number).css("margin-top", "15px");
141       $(document.body).append(div_result_number);
142 
143       var div_container_result = document.createElement("div");
144       $(div_container_result).attr("id", "container_result");
145       $(div_container_result).addClass("container");
146       $(document.body).append(div_container_result);
147 
148       var div_result = document.createElement("ul");
149       $(div_result).attr("id", "display_result");
150       $(div_result).addClass("list-group");
151       $(div_container_result).append(div_result);
152 
153       var div_container_pagination = document.createElement("div");
154       $(div_container_pagination).addClass(
155         "container d-flex justify-content-between"
156       );
157       $(div_container_pagination).attr("id", "div_container_pagination");
158       $(div_container_pagination).append(
159         '<div class="p-2 visible" style="width: 180px;"></div>'
160       );
161 
162       var div_pagination = document.createElement("ul");
163       $(div_pagination).attr("id", "display_pagination");
164       $(div_pagination).addClass("p-2 pagination ");
165       $(div_container_pagination).append($(div_pagination));
166 
167       var div_container_entrie = document.createElement("select");
168       $(div_container_entrie).attr("id", "select_entrie");
169       $(div_container_entrie).addClass("p-2 custom-select col-12 col-lg-2");
170       $(div_container_entrie).append(
171         '<option value="25">25 entries/page</option>'
172       );
173       $(div_container_entrie).append(
174         '<option value="50">50 entries/page</option>'
175       );
176       $(div_container_entrie).append(
177         '<option value="100">100 entries/page</option>'
178       );
179       $(div_container_entrie).change(function () {
180         var numentries = $(this).children("option:selected").val();
181         search(1, numentries);
182       });
183 
184       $(div_container_pagination).append($(div_container_entrie));
185       $(div_container_pagination).insertBefore($(div_result));
186 
187       $(".container").css({ width: "auto", height: "auto", display: "table" });
188 
189       // To remove the index page after first submit
190       var elem = document.getElementById("div_container_home");
191       if (elem !== null) {
192         document.getElementById("div_container_home").remove();
193       }
194     } else {
195       // To remove last result after a new submit
196       $("#display_result").empty();
197       $("#display_pagination").empty();
198       $("#div_pagination_container").remove();
199       $("#display_pagination_prime_container").remove();
200       $("#display_pagination2").empty();
201       $("#result_number").empty();
202       $("#confused_scientist_img").remove();
203       entrie_page = parseInt(
204         $("#select_entrie").children("option:selected").val()
205       );
206     }
207     // Sent parameter to ajax request
208     query = $("#query").val();
209     query_type = $("#id_typeQuery").val();
210     ajaxGet(page_number, entrie_page, query, query_type, false);
211   }
212 }

216 /**
217  * Function use to search when user access the website by url with information <br/>
218  * to read parameter in URL for window.onload -> keep_history.js
219  */
220 
221 function searchURL() {
222   let url = new URL(document.location.href);
223   let type = url.searchParams.get("type");
224   let q = url.searchParams.get("q");
225   let page = parseInt(url.searchParams.get("page"), 10);
226   let entrie = parseInt(url.searchParams.get("showresult"), 10);
227 
228   if (type != null && q != null && page != null && entrie != null) {
229     $("#query").val(q);
230     $("#id_typeQuery").val(type).change();
231     search(page, entrie);
232   }
233 }
